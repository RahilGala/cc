
import java.io.*;
import java.net.*;

class Server {
    public static void main(String[] args) throws Exception {
        ServerSocket ss = new ServerSocket(5000);
        System.out.println("Server started...");
        Socket s = ss.accept();
        System.out.println("Client connected");

        DataInputStream dis = new DataInputStream(s.getInputStream());
        DataOutputStream dos = new DataOutputStream(s.getOutputStream());

        int n=dis.readInt(); String res=""; for(int i=2;i<=n;i++){boolean p=true; for(int j=2;j<=i/2;j++)if(i%j==0)p=false; if(p)res+=i+" ";} dos.writeUTF(res);

        s.close();
        ss.close();
    }
}

class Client {
    public static void main(String[] args) throws Exception {
        Socket s = new Socket("localhost",5000);

        DataInputStream dis = new DataInputStream(s.getInputStream());
        DataOutputStream dos = new DataOutputStream(s.getOutputStream());
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        System.out.print("Enter range: "); int n=Integer.parseInt(br.readLine()); dos.writeInt(n);

        System.out.println(dis.readUTF());

        s.close();
    }
}
