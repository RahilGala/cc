
import java.io.*;
import java.net.*;

class Server {
    public static void main(String[] args) throws Exception {
        ServerSocket ss = new ServerSocket(5000);
        System.out.println("Server started...");
        Socket s = ss.accept();
        System.out.println("Client connected");

        DataInputStream dis = new DataInputStream(s.getInputStream());
        DataOutputStream dos = new DataOutputStream(s.getOutputStream());

        double r=dis.readDouble(),h=dis.readDouble(); dos.writeDouble(2*Math.PI*r*(r+h));

        s.close();
        ss.close();
    }
}

class Client {
    public static void main(String[] args) throws Exception {
        Socket s = new Socket("localhost",5000);

        DataInputStream dis = new DataInputStream(s.getInputStream());
        DataOutputStream dos = new DataOutputStream(s.getOutputStream());
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        System.out.println("Enter r and h:"); double r=Double.parseDouble(br.readLine()); double h=Double.parseDouble(br.readLine()); dos.writeDouble(r); dos.writeDouble(h);

        System.out.println(dis.readDouble());

        s.close();
    }
}
